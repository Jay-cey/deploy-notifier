// install: npm i express @octokit/rest @slack/web-api @google/generative-ai dotenv
require('dotenv').config();
const express  = require('express');
const { Octokit }   = require('@octokit/rest');
const { WebClient } = require('@slack/web-api');
const { GoogleGenerativeAI } = require('@google/generative-ai');

const app     = express();
const octokit = new Octokit({ auth: process.env.GH_TOKEN });
const slack   = new WebClient(process.env.SLACK_TOKEN);
const genAI   = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);
app.use(express.json());

app.get('/', (req, res) => res.status(200).send('Webhook server is running'));

app.post('/webhook/github', async (req, res) => {
  try {
    const { ref, head_commit, repository } = req.body;
    if (!ref || !repository) return res.sendStatus(400);
    const defaultBranch = repository.default_branch || 'main';
    if (ref !== 'refs/heads/' + defaultBranch) return res.sendStatus(200);
    if (!head_commit) return res.sendStatus(400);

    const [owner, repo] = repository.full_name.split('/');
    const sha = head_commit.id;

    const { data } = await octokit.repos.getCommit({ owner, repo, ref: sha });
    const diffText = data.files
      .map(f => `${f.filename} (+${f.additions} -${f.deletions})\n${f.patch || ''}`)
      .join('\n\n');

    const model = genAI.getGenerativeModel({ model: 'gemini-2.5-flash' });
    const result = await model.generateContent(
      `Summarize this diff in 2-3 sentences for a Slack deploy alert.\n\n${diffText}`
    );
    const summary = result.response.text();

    const files = data.files.slice(0, 5)
      .map(f => `+${f.additions} -${f.deletions}  ${f.filename}`).join('\n');
    const extra = data.files.length - 5;
    const filesText = extra > 0 ? `${files}\n... and ${extra} more` : files;

    await slack.chat.postMessage({
      channel: process.env.SLACK_CHANNEL,
      text: `🚀 *Deployed* · \`${sha.slice(0,7)}\``,
      blocks: [{ type: 'section', text: { type: 'mrkdwn',
        text: `🚀 *Deployed to production* · \`${sha.slice(0,7)}\`\n*${head_commit.message}* — ${head_commit.author.name}\n\n${summary}\n\`\`\`${filesText}\`\`\`` }
      }]
    });

    res.sendStatus(200);
  } catch (err) {
    console.error(err);
    res.status(500).send(err.message);
  }
});

if (require.main === module) {
  const PORT = process.env.PORT || 3000;
  app.listen(PORT, () => console.log(`Listening on :${PORT}`));
}
module.exports = { app, octokit, slack, genAI };
